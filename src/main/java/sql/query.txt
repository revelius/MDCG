create database mg;

use mg;

/*자유게시판 제작용*/
create table free_board(
	id integer(4) primary key auto_increment,
	name varchar(10),
	title varchar(50),
	content text,
	date date,
	hit Integer(4),
	cgroup Integer(4),
	step Integer(4),
	indent Integer(4)
);
